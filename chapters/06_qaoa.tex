\section{QAOA и MaxCut}
\label{sec:qaoa}

% --- Увод у QAOA ---
% QAOA (Quantum Approximate Optimization Algorithm) предложили су
% Farhi, Goldstone и Gutmann (2014). То је хибридни квантно-класични
% варијациони алгоритам за комбинаторну оптимизацију.

\subsection{QUBO формулација MaxCut-а}

% --- QUBO ---
% Пре примене QAOA, потребно је MaxCut формулисати као QUBO
% (Quadratic Unconstrained Binary Optimization) проблем.
% Уведите бинарне варијабле z_v ∈ {-1, +1} (спинове).
% Ивица (u,v) прелази рез ако z_u ≠ z_v, тј. z_u·z_v = -1.

Уведимо спин варијабле $z_v \in \{-1, +1\}$ за сваки чвор $v \in V$.
Ивица $\{u,v\}$ прелази рез ако и само ако $z_u z_v = -1$, тј.
$\frac{1 - z_u z_v}{2} = 1$.
Циљна функција постаје:
\[
  C(\mathbf{z}) = \sum_{\{u,v\} \in E} \frac{1 - z_u z_v}{2}.
\]

Максимизација $C(\mathbf{z})$ еквивалентна је минимизацији
$-C(\mathbf{z})$, што је проблем налажења основног стања
Изинговог Хамилтонијана:
\[
  \hat{H}_C = \frac{1}{2}\sum_{\{u,v\} \in E} \bigl(\hat{I} - \hat{Z}_u \hat{Z}_v\bigr).
\]

\subsection{Структура QAOA кола}

% --- QAOA коло ---
% QAOA глубине p примењује наизменично:
%   1. Фазни оператор: e^{-iγ Ĥ_C} (кодира циљну функцију)
%   2. Мешачки оператор: e^{-iβ Ĥ_B}, Ĥ_B = Σ_v X̂_v
% Почетно стање: равномерна суперпозиција |+⟩^⊗n = H^⊗n |0⟩^⊗n.

\begin{algorithm}
\caption{QAOA за MaxCut, дубина $p$}
\begin{algorithmic}[1]
\State Припреми почетно стање: $|\psi_0\rangle = H^{\otimes n}|0\rangle^{\otimes n}$
\For{$k = 1, \ldots, p$}
  \State Примени фазни оператор: $e^{-i\gamma_k \hat{H}_C}$
  \State Примени мешачки оператор: $e^{-i\beta_k \hat{H}_B}$
\EndFor
\State Измери стање и израчунај $\langle \hat{H}_C \rangle$
\State Класично оптимизуј параметре $(\boldsymbol{\gamma}, \boldsymbol{\beta})$
       (нпр. градијентним спустом или COBYLA)
\State \Return најбоља измерена конфигурација $\mathbf{z}^*$
\end{algorithmic}
\end{algorithm}

\subsection{Теоријски резултати за p=1}

% --- p=1 аналитика ---
% За p=1 постоји аналитичко решење за очекивану вредност
% циљне функције. За 3-регуларне графове, QAOA са p=1 постиже
% апроксимациони однос ≥ 0.6924 (Farhi et al. 2014).

За $p = 1$ и 3-регуларне графове, QAOA постиже апроксимациони однос:
\[
  \frac{\langle \hat{H}_C \rangle_{\gamma^*, \beta^*}}{|E|} \;\geq\; 0.6924.
\]
